### 如何计算整数二进制的个数

1. `n & (n - 1)` 将二进制最后一位置为 0

```js
function countBits(n) {
  let count = 0;
  while (n) {
    n = n & (n - 1);
    count++;
  }
  return count;
}
```

2. 通过内置 API 实现

```js
function countBits(n) {
  return n.toString(2).split('0').join('').length;
}
```

3. 二进制与 1 相与

```js
function countBits(n) {
  let count = 0;
  while ((n = n >> 1)) {
    count += n & 1;
  }
}
```
