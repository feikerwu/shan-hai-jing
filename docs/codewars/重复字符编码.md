1. 通过计算每个字符出现次数，时间复杂度 $O(n)$, 空间复杂度 $O(n)$

```js
function duplicateEncode(word) {
  word = word.toLowerCase();
  let size = word.split('').reduce((acc, cur) => {
    if (!acc[cur]) {
      acc[cur] = 0;
    }
    acc[cur] = acc[cur] + 1;
    return acc;
  }, {});
  return word
    .split('')
    .map((item) => (size[item] > 1 ? ')' : '('))
    .join('');
}
```

2. 通过判断第一次出现和最后一次出现的位置来区分

```js
function duplicateEncode(word) {
  return word
    .toLowerCase()
    .split('')
    .map((item, index, arr) =>
      arr.indexOf(item) !== arr.lastIndexOf(item) ? ')' : '('
    )
    .join('');
}
```
